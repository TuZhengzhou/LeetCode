iota 函数

    函数原型
        template <class ForwardIterator, class T>
        void iota (ForwardIterator first, ForwardIterator last, T val)
        {
            while (first!=last) {
                *first = val;
                ++first;
                ++val;
            }
        }

    用法
        vector<int> a = vector<int>(n);
        std::iota (a.begin(), a.end(), 100);

        int numbers[10];
        std::iota (numbers,numbers+10,100);
